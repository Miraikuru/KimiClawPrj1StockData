# Nyaron Warriors - 喵灵战士
# 基因改造的战斗特化单位

nyaron_warriors = {
	# 基础属性
	type = nyaron_warriors
	
	# 购买和维护成本
	cost = 150
	maintenance = 1.5
	
	# 战斗属性
	damage = 35
	toughness = 25
	pursuit = 40
	screen = 30
	
	# 地形加成
	terrain_bonus = {
		forest = { damage = 10 pursuit = 10 }
		mountains = { toughness = 10 screen = 10 }
		hills = { damage = 5 toughness = 5 }
	}
	
	# 克制关系
	counters = {
		light_cavalry = 1.5
		archers = 1.3
	}
	
	# 被克制
	weak_against = {
		pikemen = 1.5
		heavy_cavalry = 1.3
	}
	
	# 购买限制
	max_manpower = 10
	
	# 招募条件
	can_recruit = {
		culture = { has_culture_tradition = tradition_stellar_origins }
		OR = {
			has_trait = nyaron
			has_trait = nyaron_pureblood
		}
	}
	
	# 维护条件
	can_maintain = {
		OR = {
			has_trait = nyaron
			has_trait = nyaron_pureblood
		}
	}
	
	# 购买效果
	on_bought = {
		add_prestige = 50
	}
	
	# 图标
	icon = nyaron_warriors_icon
}
